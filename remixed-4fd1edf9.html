function updateMcBudget(){
  mcBudget+=0.0008;
  const pct=Math.min(mcBudget/mcBudgetCap*100,100);
  const fill=document.getElementById('mc-bfill');const txt=document.getElementById('mc-btxt');
  if(fill){fill.style.width=pct+'%';fill.style.background=pct>80?'var(--red)':pct>60?'var(--yellow)':'var(--green)';}
  if(txt){txt.textContent=`${mcBudget.toFixed(2)}/${mcBudgetCap.toFixed(2)}`;txt.style.color=pct>80?'var(--red)':pct>60?'var(--yellow)':'var(--green)';}
  if(pct>=100&&!mcStopped){toggleMcStop();mcLog('mle','ZEUS','Budget cap — STOP automático');showToast('⚠️ Budget cap alcanzado');}
}
